# main.py
from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel
from supabase import create_client, Client
import os

app = FastAPI()

# Config
url: str = os.environ.get("SUPABASE_URL")
key: str = os.environ.get("SUPABASE_KEY")
supabase: Client = create_client(url, key)

class TransferRequest(BaseModel):
    sender_id: str
    receiver_id: str
    amount: float

@app.post("/api/transfer")
async def transfer_credits(tx: TransferRequest):
    # 1. AI Fraud Check (Simplified)
    if tx.amount > 5000:
        # Trigger Manual Review
        return {"status": "pending_review", "reason": "High Value"}

    # 2. Execute Transaction (Atomic)
    try:
        # Call Supabase Stored Procedure to ensure ACID compliance
        response = supabase.rpc('transfer_funds', {
            'p_sender': tx.sender_id,
            'p_receiver': tx.receiver_id,
            'p_amount': tx.amount
        }).execute()
        return {"status": "success", "tx_id": response.data}
    except Exception as e:
        raise HTTPException(status_code=400, detail=str(e))
