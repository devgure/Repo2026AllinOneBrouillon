from langchain_core.tools import tool

# State schema
class AgentState(TypedDict):
    claim_id: str
    denial_reason: str
    confidence: float
    action: str

# Tools
@tool
def fetch_payer_policy(payer: str, cpt_code: str) -> dict:
    # Scrape CMS/payer site
    return {"policy": "Modifier 59 required"}

@tool
def generate_appeal_letter(denial: str, policy: dict) -> str:
    return f"Per {policy['policy']}, we appeal..."

# Agent logic
def plan_node(state: AgentState) -> AgentState:
    state["action"] = "fetch_policy"
    return state

def execute_node(state: AgentState) -> AgentState:
    if state["action"] == "fetch_policy":
        policy = fetch_payer_policy("UHC", "70553")
        letter = generate_appeal_letter(state["denial_reason"], policy)
        state["confidence"] = 0.92  # < 0.95 → HITL
    return state

# Build graph
workflow = StateGraph(AgentState)
workflow.add_node("plan", plan_node)
workflow.add_node("execute", execute_node)
workflow.set_entry_point("plan")
workflow.add_edge("plan", "execute")
workflow.add_edge("execute", END)

app = workflow.compile()
